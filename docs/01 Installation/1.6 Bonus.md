# 1.6 Bonus

Installation automatisée de Alpine Linux

cat > /root/tp-answers.conf <<'EOF'
# TP: 1.1 Installation d’Alpine Linux (valeurs recommandées)

```
# Clavier
KEYMAPOPTS="fr fr-nodeadkeys"

# Nom machine
HOSTNAMEOPTS="alpine-TP"

# Gestionnaire de périphériques
DEVDOPTS="mdev"

# Réseau (DHCP sur eth0 + eth1)
INTERFACESOPTS="auto lo
iface lo inet loopback

auto eth0
iface eth0 inet dhcp

auto eth1
iface eth1 inet dhcp
"

# Timezone
TIMEZONEOPTS="Europe/Paris"

# Pas de proxy
PROXYOPTS="none"

# Miroir APK : plus rapide + community
APKREPOSOPTS="non"

# Création utilisateur classique "user"
# (-a = admin via wheel, -u = crée l'utilisateur)
USEROPTS="-a -u user"

# SSH
SSHDOPTS="openssh"

# NTP
NTPOPTS="chrony"

# Installation disque : /dev/sda en mode sys (persistant)
DISKOPTS="-m sys /dev/sda"
```


```
setup-alpine -f /root/tp-answers.conf
```
